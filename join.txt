SELECT DISTINCT
    klienci.imie_klienta,
    klienci.nazwisko_klienta,
    samochody.marka,
    samochody.model
FROM klienci 
	INNER JOIN wypozyczenia 
    	ON klienci.id_klienta = wypozyczenia.id_klienta
	INNER JOIN dane_wypozyczen
 		on dane_wypozyczen.id_wypozyczenia = wypozyczenia.id_wypozyczenia
    INNER JOIN samochody
    	on samochody.id_samochodu = dane_wypozyczen.id_samochodu;




2. 
SELECT DISTINCT
	klienci.imie_klienta,
    klienci.nazwisko_klienta,
    pracownicy.imie_pracownika,
    pracownicy.nazwisko_pracownika
FROM klienci
	INNER JOIN wypozyczenia
    	ON klienci.id_klienta = wypozyczenia.id_klienta
    INNER JOIN pracownicy
    	ON pracownicy.id_pracownika = wypozyczenia.id_pracownika
WHERE klienci.miasto_klienta LIKE "Gliwice";

4.
SELECT 
	samochody.id_samochodu,
    samochody.marka,
    samochody.model,
    samochody.nr_rejestracyjny,
    klasy_samochodow.nazwa_klasy
FROM samochody
	INNER JOIN klasy_samochodow
    	ON klasy_samochodow.id_klasy = samochody.id_klasy
    INNER JOIN dane_wypozyczen
    	ON dane_wypozyczen.id_samochodu = samochody.id_samochodu
    INNER JOIN wypozyczenia
    	ON wypozyczenia.id_wypozyczenia = dane_wypozyczen.id_wypozyczenia
WHERE wypozyczenia.data_wyp BETWEEN '2022-12-24' AND '2022-12-26';
